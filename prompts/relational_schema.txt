users {
	user_id int pk
	address_id int > address.address_id
	first_name varchar
	last_name varchar
	email varchar
	password varchar
}

address {
	address_id int pk
	street varchar
	city varchar
	state varchar
	country varchar
	zip int
}

session {
	session_id int pk
	user_id int > users.user_id
	device_type varchar
	created_at timestamp
	expires_at timestamp
	is_active boolean
}

cart {
	cart_id int pk
	user_id int > users.user_id
	status enum
	created_at timestamp
	update_at timestamp
}

cart_items {
	cart_item_id int pk
	cart_id int > cart.cart_id
	quantity int
	added_at timestamp
	sku int > product_variant.sku
}

product_variant {
	sku int pk
	product_id int > product.product_id
	price double
	status enum
	stock_quantity int
}

product {
	name varchar
	description varchar
	brand varchar
	manufacturer varchar
	product_id int pk
	category_id int > categories.category_id
}
